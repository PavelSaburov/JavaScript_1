var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2 => префиксная форма записи, увеличение происходит до присваивания, а=2
d = b++; alert(d); // 1 => постфиксная форма записи, увеличение происходит после присваивания, b=2
c = (2+ ++a); alert(c); // 5 => происходит увеличение а на 1 с 2 до 3, затем прибавляется 2, а=3
d = (2+ b++); alert(d); // 4 => происходит сложение переменной b и числа 2, затем b увеличивается на 1, b=3
alert(a); // 3
alert(b); // 3
